 {

     if (product.Count >= 0)
     {
         product.Add(new Products { PId = Guid.NewGuid().ToString() , PImage = "cup.png", PName = $"{PNamelbl.Text}", PPrice = $"{PPricelbl.Text}" });
     }

     product.ToList();

     lblone.Text = product.Count.ToString();

     SemanticScreenReader.Announce(lblone.Text);

 ObservableCollection<Products> cartitemlist = new ObservableCollection<Products>();

 public CartPage()
	{
		InitializeComponent();

     cartitemlist = JsonConvert.DeserializeObject<ObservableCollection<Products>>(Cartlist);    

     collectionviewlist.ItemsSource = cartitemlist;
    
 }

 private void SwipeItem_Invoked(object sender, EventArgs e)
 {
     DisplayAlert("Alert", "Are You Sure To Delete This item", "Ok");
     cartitemlist.ToList();

    // object value = cartitemlist.FirstOrDefault(x => x.PName == namelbl.Text);

 }

 private void RemoveBtn_Clicked(object sender, EventArgs e)
 {
